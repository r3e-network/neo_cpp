414 static std::string ValidateHexString(const std::string& hex, size_t expectedLength = 0)
415 {
416 if (hex.empty())
417 {
418 throw std::runtime_error("Hex string cannot be empty");
419 }
420
421 // Remove 0x prefix if present
422 std::string cleanHex = hex;
423 if (cleanHex.length() >= 2 && cleanHex[0] == '0' && (cleanHex[1] == 'x' || cleanHex[1] == 'X'))
424 {
425 cleanHex = cleanHex.substr(2);
426 }
427
428 if (cleanHex.empty())
429 {
430 throw std::runtime_error("Hex string cannot be just '0x'");
431 }
432
433 // Must have even number of hex digits
434 if (cleanHex.length() % 2 != 0)
435 {
436 throw std::runtime_error("Hex string must have even number of digits");
437 }
438
439 // Check expected length
440 if (expectedLength > 0 && cleanHex.length() / 2 != expectedLength)
441 {
442 throw std::runtime_error("Hex string wrong length: expected " + std::to_string(expectedLength) +
443 " bytes, got " + std::to_string(cleanHex.length() / 2));
444 }
445
446 // Validate hex characters
447 for (char c : cleanHex)
448 {
449 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F')))
450 {
451 throw std::runtime_error("Invalid hex character: " + std::string(1, c));
452 }
453 }
454
455 return cleanHex;
456 }
