127class Histogram : public Metric
128{
129 public:
130 Histogram(const std::string& name, const std::string& help,
131 const std::vector<double>& buckets = {0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5, 10})
132 : Metric(name, help, MetricType::HISTOGRAM), buckets_(buckets)
133 {
134 // Initialize bucket counters
135 for (size_t i = 0; i <= buckets.size(); ++i)
136 {
137 bucketCounts_.push_back(0);
138 }
139 }
140
141 void Observe(double value)
142 {
143 std::lock_guard<std::mutex> lock(mutex_);
144
145 sum_ += value;
146 count_++;
147
148 // Find bucket
149 size_t i = 0;
150 for (; i < buckets_.size(); ++i)
151 {
152 if (value <= buckets_[i])
153 {
154 break;
155 }
156 }
157
158 // Increment all buckets >= value
159 for (size_t j = i; j <= buckets_.size(); ++j)
160 {
161 bucketCounts_[j]++;
162 }
163 }
164
165 std::string Serialize() const override
166 {
167 std::lock_guard<std::mutex> lock(mutex_);
168
169 std::stringstream ss;
170 ss << "# HELP " << name_ << " " << help_ << "\n";
171 ss << "# TYPE " << name_ << " " << TypeString() << "\n";
172
173 // Buckets
174 for (size_t i = 0; i < buckets_.size(); ++i)
175 {
176 ss << name_ << "_bucket{le=\"" << buckets_[i] << "\"} " << bucketCounts_[i] << "\n";
177 }
178 ss << name_ << "_bucket{le=\"+Inf\"} " << bucketCounts_[buckets_.size()] << "\n";
179
180 // Sum and count
181 ss << name_ << "_sum " << sum_ << "\n";
182 ss << name_ << "_count " << count_ << "\n";
183
184 return ss.str();
185 }
186
187 private:
188 std::vector<double> buckets_;
189 std::vector<uint64_t> bucketCounts_;
190 double sum_ = 0;
191 uint64_t count_ = 0;
192 mutable std::mutex mutex_;
193};

236

238 bool first = true;
239 for (const auto& [key, entry] : metrics_)
240 {
241 if (first)
242 {
243 ss << "# HELP " << name_ << " " << help_ << "\n";
244 ss << "# TYPE " << name_ << " " << entry.metric->TypeString() << "\n";

246 }
