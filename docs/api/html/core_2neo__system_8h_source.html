197 template <typename T>
198 std::shared_ptr<T> get_service(std::function<bool(const T&)> filter = nullptr) const
199 {
200 std::lock_guard<std::mutex> lock(services_mutex_);
201 for (const auto& service : services_)
202 {
203 if (auto typed_service = std::dynamic_pointer_cast<T>(service))
204 {
205 if (!filter || filter(*typed_service))
206 {
207 return typed_service;
208 }
209 }
210 }
211 return nullptr;
212 }
