66 std::shared_ptr<Connection> Acquire()
67 {
68 std::unique_lock<std::mutex> lock(mutex_);
69
70 auto deadline = std::chrono::steady_clock::now() + config_.connectionTimeout;
71
72 while (running_)
73 {
74 // Try to get an available connection
75 if (!available_.empty())
76 {
77 auto conn = available_.front();
78 available_.pop();
79
80 // Validate connection before returning
81 if (validator_(conn))
82 {
83 active_[conn.get()] = std::chrono::steady_clock::now();
84 return conn;
85 }
86 // Invalid connection, continue to create new one
87 }
88
89 // Check if we can create a new connection
90 if (available_.size() + active_.size() < config_.maxConnections)
91 {
92 lock.unlock();
93
94 try
95 {
96 auto conn = factory_();
97 if (conn && validator_(conn))
98 {
99 lock.lock();
100 active_[conn.get()] = std::chrono::steady_clock::now();
101 return conn;
102 }
103 }
104 catch (const std::exception&)
105 {
106 // Failed to create connection
107 }
108
109 lock.lock();
110 }
111
112 // Wait for a connection to become available
113 if (cv_.wait_until(lock, deadline) == std::cv_status::timeout)
114 {
115 return nullptr; // Timeout
116 }
117 }
118
119 return nullptr;

326 void WorkerLoop()
327 {
328 std::unique_lock<std::mutex> lock(mutex_);
329
330 while (running_)
331 {
332 if (timeouts_.empty())
333 {
334 cv_.wait(lock);
335 continue;
336 }
337
338 // Find next timeout
339 auto now = std::chrono::steady_clock::now();
340 std::vector<uint64_t> expired;
341
342 for (const auto& [id, entry] : timeouts_)
343 {
344 if (entry.deadline <= now)
345 {
346 expired.push_back(id);
347 }
348 }
349
350 // Execute expired timeouts
351 for (uint64_t id : expired)
352 {
353 auto it = timeouts_.find(id);
354 if (it != timeouts_.end())
355 {
356 auto callback = it->second.callback;
357 timeouts_.erase(it);
358
359 // Execute callback without holding lock
360 lock.unlock();
361 callback();
362 lock.lock();
363 }
364 }
365
366 // Wait until next timeout or notification
367 if (!timeouts_.empty())
368 {
369 auto nextDeadline =
370 std::min_element(timeouts_.begin(), timeouts_.end(),
371 [](const auto& a, const auto& b) { return a.second.deadline < b.second.deadline; })
372 ->second.deadline;
373
374 cv_.wait_until(lock, nextDeadline);
375 }
376 }
377 }
