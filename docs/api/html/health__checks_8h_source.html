143 {
144 auto start = std::chrono::steady_clock::now();
145 HealthCheckResult result;
146 result.details["type"] = "disk_space";
147 result.details["path"] = path_;
148
149 try
150 {
151 auto space = std::filesystem::space(path_);
152 uint64_t free_bytes = space.available;
153 uint64_t total_bytes = space.capacity;
154 double free_percentage = (double)free_bytes / total_bytes * 100.0;
155
156 result.details["free_gb"] = std::to_string(free_bytes / (1024ULL * 1024ULL * 1024ULL));
157 result.details["total_gb"] = std::to_string(total_bytes / (1024ULL * 1024ULL * 1024ULL));
158 result.details["free_percentage"] = std::to_string(free_percentage);
159
160 if (free_bytes < min_free_bytes_)
161 {
162 result.status = HealthStatus::Unhealthy;
163 result.message = "Critical: Low disk space";
164 }
165 else if (free_bytes < warning_free_bytes_)
166 {
167 result.status = HealthStatus::Degraded;
168 result.message = "Warning: Disk space running low";
169 }
170 else
171 {
172 result.status = HealthStatus::Healthy;
173 result.message = "Disk space adequate";
174 }
175 }
176 catch (const std::exception& e)
