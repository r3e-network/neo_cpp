83 template <typename T>
84 static RecoveryResult<T> Retry(std::function<T()> operation, const RetryConfig& config = RetryConfig())
85 {
86 std::string last_error;
87 NeoException::ErrorCode last_error_code = NeoException::ErrorCode::UNKNOWN_ERROR;
88
89 for (int attempt = 1; attempt <= config.max_attempts; ++attempt)
90 {
91 try
92 {
93 T result = operation();
94 return RecoveryResult<T>(result, attempt);
95 }
96 catch (const NeoException& e)
97 {
98 last_error = e.what();
99 last_error_code = e.GetErrorCode();
100
101 if (!config.should_retry(e) || attempt == config.max_attempts)
102 {
103 break;
104 }
105 }
106 catch (const std::exception& e)
107 {
108 last_error = e.what();
109
110 if (!config.should_retry(e) || attempt == config.max_attempts)
111 {
112 break;
113 }
114 }
115
116 // Calculate delay for next attempt
117 if (attempt < config.max_attempts)
118 {
119 auto delay = config.base_delay;
120 if (config.exponential_backoff)
121 {
122 for (int i = 1; i < attempt; ++i)
123 {
124 delay = std::chrono::milliseconds(
125 static_cast<long long>(delay.count() * config.backoff_multiplier));
126 }
127 }
128 delay = std::min(delay, config.max_delay);
129 std::this_thread::sleep_for(delay);
130 }
131 }
132
133 return RecoveryResult<T>(last_error, last_error_code, config.max_attempts);
134 }

183 class CircuitBreaker
184 {
185 public:
186 enum class State
187 {
188 CLOSED, // Normal operation
189 OPEN, // Circuit is open, calls are failing fast
190 HALF_OPEN // Testing if service has recovered
191 };
192
193 struct Config
194 {
195 int failure_threshold = 5; // Failures before opening circuit
196 std::chrono::seconds timeout{30}; // Time before trying half-open
197 int success_threshold = 2; // Successes needed to close circuit
198 };
199
200 explicit CircuitBreaker(const Config& config) : config_(config) {}
201 CircuitBreaker() : CircuitBreaker(Config{}) {}
202
203 template <typename T>
204 RecoveryResult<T> Execute(std::function<T()> operation)
205 {
206 std::lock_guard<std::mutex> lock(mutex_);
207
208 if (state_ == State::OPEN)
209 {
210 if (std::chrono::steady_clock::now() - last_failure_time_ > config_.timeout)
211 {
212 state_ = State::HALF_OPEN;
213 consecutive_successes_ = 0;
214 }
215 else
216 {
217 return RecoveryResult<T>("Circuit breaker is OPEN", NeoException::ErrorCode::TIMEOUT, 0);
218 }
219 }
220
221 try
222 {
223 T result = operation();
224 OnSuccess();
225 return RecoveryResult<T>(result);
226 }
227 catch (const NeoException& e)
228 {
229 OnFailure();
230 return RecoveryResult<T>(e.what(), e.GetErrorCode(), 1);
231 }
232 catch (const std::exception& e)
233 {
234 OnFailure();
235 return RecoveryResult<T>(e.what(), NeoException::ErrorCode::UNKNOWN_ERROR, 1);
236 }
237 }
238
239 State GetState() const { return state_; }
240 int GetFailureCount() const { return failure_count_; }
241
242 private:
243 Config config_;
244 State state_ = State::CLOSED;
245 int failure_count_ = 0;
246 int consecutive_successes_ = 0;
247 std::chrono::steady_clock::time_point last_failure_time_;
248 mutable std::mutex mutex_;
249
250 void OnSuccess()
251 {
252 failure_count_ = 0;
253 if (state_ == State::HALF_OPEN)
254 {
255 consecutive_successes_++;
256 if (consecutive_successes_ >= config_.success_threshold)
257 {
258 state_ = State::CLOSED;
259 }
260 }
261 }
262
263 void OnFailure()
264 {
265 failure_count_++;
266 last_failure_time_ = std::chrono::steady_clock::now();
267 consecutive_successes_ = 0;
268
269 if (failure_count_ >= config_.failure_threshold)
270 {
271 state_ = State::OPEN;
272 }
273 }

291 template <typename T>
292 RecoveryResult<T> Execute(std::function<T()> operation)
293 {
294 std::unique_lock<std::mutex> lock(mutex_);
295
296 if (current_calls_ >= config_.max_concurrent_calls)
297 {
298 return RecoveryResult<T>("Bulkhead capacity exceeded", NeoException::ErrorCode::TIMEOUT, 0);
299 }
300
301 current_calls_++;
302 lock.unlock();
303
304 try
305 {
306 T result = operation();
307
308 lock.lock();
309 current_calls_--;
310 return RecoveryResult<T>(result);
311 }
312 catch (const NeoException& e)
313 {
314 lock.lock();
315 current_calls_--;
316 return RecoveryResult<T>(e.what(), e.GetErrorCode(), 1);
317 }
318 catch (const std::exception& e)
319 {
320 lock.lock();
321 current_calls_--;
322 return RecoveryResult<T>(e.what(), NeoException::ErrorCode::UNKNOWN_ERROR, 1);
323 }
324 }

337 template <typename T>
338 static RecoveryResult<T> WithTimeout(std::function<T()> operation, std::chrono::milliseconds timeout)
339 {
340 std::promise<RecoveryResult<T>> promise;
341 auto future = promise.get_future();
342
343 std::thread worker(
344 [&promise, operation]()
345 {
346 try
347 {
348 T result = operation();
349 promise.set_value(RecoveryResult<T>(result));
350 }
351 catch (const NeoException& e)
352 {
353 promise.set_value(RecoveryResult<T>(e.what(), e.GetErrorCode(), 1));
354 }
355 catch (const std::exception& e)
356 {
357 promise.set_value(RecoveryResult<T>(e.what(), NeoException::ErrorCode::UNKNOWN_ERROR, 1));
358 }
359 });
360
361 if (future.wait_for(timeout) == std::future_status::timeout)
362 {
363 // Worker thread is detached on timeout to prevent blocking
364 worker.detach();
365 return RecoveryResult<T>("Operation timed out", NeoException::ErrorCode::TIMEOUT, 1);
366 }
367
368 worker.join();
369 return future.get();
370 }
