259 std::string SubstituteEnvironmentVariables(const std::string& str)
260 {
261 std::regex envRegex(R"(\$\{([A-Z_][A-Z0-9_]*)\})");
262 std::string result = str;
263
264 std::smatch match;
265 while (std::regex_search(result, match, envRegex))
266 {
267 std::string varName = match[1].str();
268 const char* envValue = std::getenv(varName.c_str());
269
270 if (envValue)
271 {
272 result.replace(match.position(), match.length(), envValue);
273 }
274 else
275 {
276 // Keep the variable reference if env var not found
277 break;
278 }
279 }
280
281 return result;
282 }
