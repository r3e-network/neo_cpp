27 static bool IsValidHexString(const std::string& hex, size_t expectedLength = 0)
28 {
29 if (hex.empty()) return false;
30
31 // Check if starts with 0x
32 size_t start = 0;
33 if (hex.size() >= 2 && hex[0] == '0' && (hex[1] == 'x' || hex[1] == 'X'))
34 {
35 start = 2;
36 }
37
38 // Must have even number of hex digits
39 size_t hexLength = hex.length() - start;
40 if (hexLength % 2 != 0) return false;
41
42 // Check expected length
43 if (expectedLength > 0 && hexLength / 2 != expectedLength)
44 {
45 return false;
46 }
47
48 // Validate hex characters
49 for (size_t i = start; i < hex.length(); i++)
50 {
51 char c = hex[i];
52 if (!((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F')))
53 {
54 return false;
55 }
56 }
57
58 return true;
59 }

301 static void ValidateGasAmount(const std::string& gas)
302 {
303 try
304 {
305 double amount = core::SafeConversions::SafeToDouble(gas);
306 if (amount < 0)
307 {
308 throw std::invalid_argument("Gas amount cannot be negative");
309 }
310 if (amount > 10000000000)
311 { // 10 billion GAS max
312 throw std::invalid_argument("Gas amount too large");
313 }
314
315 // Validate decimal places (max 8 for GAS)
316 std::size_t decimal_pos = gas.find('.');
317 if (decimal_pos != std::string::npos)
318 {
319 std::size_t decimals = gas.length() - decimal_pos - 1;
320 if (decimals > 8)
321 {
322 throw std::invalid_argument("GAS amount has too many decimal places (max 8)");
323 }
324 }
325 }
326 catch (const std::runtime_error& e)
327 {
328 throw std::invalid_argument("Invalid gas amount: " + std::string(e.what()));
329 }
