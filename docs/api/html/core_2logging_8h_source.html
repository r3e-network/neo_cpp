182void FormatString(std::stringstream& ss, const std::string& fmt, T&& value)
183{
184 size_t pos = fmt.find("{}");
185 if (pos != std::string::npos)
186 {
187 ss << fmt.substr(0, pos) << value << fmt.substr(pos + 2);
188 }
189 else
190 {
191 ss << fmt << " " << value;
192 }

195template <typename T, typename... Args>
196void FormatString(std::stringstream& ss, const std::string& fmt, T&& value, Args&&... args)
197{
198 size_t pos = fmt.find("{}");
199 if (pos != std::string::npos)
200 {
201 ss << fmt.substr(0, pos) << value;
202 FormatString(ss, fmt.substr(pos + 2), std::forward<Args>(args)...);
203 }
