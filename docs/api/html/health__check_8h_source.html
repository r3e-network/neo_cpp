104 {

128 result.details["syncProgress"] = std::to_string((height * 100.0) / headerHeight) + "%";
129 }
130 else
131 {
132 result.status = HealthStatus::HEALTHY;
133 result.message = "Blockchain synced";
134 }
135 }
136 catch (const std::exception& e)
137 {
138 result.status = HealthStatus::UNHEALTHY;
139 result.message = "Failed to check blockchain: " + std::string(e.what());
140 }
141
142 auto end = std::chrono::steady_clock::now();
143 result.responseTime = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);

145 return result;
146 }
