Hello Neo - Your First Program

A simple program to connect to the Neo network and get basic information.

#include <neo/sdk/rpc/rpc_client.h>
#include <iostream>

int main() {
    try {
        // Connect to Neo testnet
        neo::sdk::RpcClient client("http://seed1.neo.org:20332");
        
        // Get blockchain height
        auto height = client.GetBlockCount();
        std::cout << "Hello Neo!" << std::endl;
        std::cout << "Current blockchain height: " << height << std::endl;
        
        // Get network version
        auto version = client.GetVersion();
        std::cout << "Network: " << version.Network << std::endl;
        std::cout << "Protocol: " << version.Protocol << std::endl;
        
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
    
    return 0;
}

Compile and Run

# Compile
g++ -std=c++20 hello_neo.cpp -lneo-sdk -o hello_neo

# Run
./hello_neo

Creating and Managing Wallets

Create a NEP-6 compliant wallet with multiple accounts.

#include <neo/sdk/wallet/wallet.h>
#include <iostream>
#include <iomanip>

int main() {
    // Create a new wallet
    neo::sdk::Wallet wallet("MyWallet");
    
    // Create multiple accounts
    std::cout << "Creating wallet with 3 accounts..." << std::endl;
    
    for (int i = 1; i <= 3; i++) {
        auto account = wallet.CreateAccount("Account " + std::to_string(i));
        
        std::cout << "\nAccount " << i << ":" << std::endl;
        std::cout << "  Address: " << account.Address() << std::endl;
        std::cout << "  Public Key: " << account.GetPublicKey().ToString() << std::endl;
        
        // Export private key (WIF format)
        auto wif = account.Export();
        std::cout << "  WIF: " << wif << std::endl;
    }
    
    // Set default account
    auto accounts = wallet.GetAccounts();
    wallet.SetDefaultAccount(accounts[0]);
    
    // Save encrypted wallet
    std::string password = "MySecurePassword123!";
    wallet.Save("wallet.json", password);
    std::cout << "\nWallet saved to wallet.json" << std::endl;
    
    // Load wallet
    auto loaded = neo::sdk::Wallet::Load("wallet.json", password);
    std::cout << "Wallet loaded successfully with " 
              << loaded.GetAccounts().size() << " accounts" << std::endl;
    
    return 0;
}

Sending NEO Tokens

Build and send a transaction to transfer NEO tokens.

#include <neo/sdk/sdk.h>
#include <iostream>

int main() {
    try {
        // Connect to network
        auto client = std::make_shared<neo::sdk::RpcClient>("http://seed1.neo.org:20332");
        
        // Load wallet
        auto wallet = neo::sdk::Wallet::Load("wallet.json", "password");
        auto account = wallet.GetDefaultAccount();
        
        // Recipient address
        std::string recipient = "NXjtqYERuvSWGawjVux8UerNejvwdYg7eE";
        
        // Check balance first
        auto balance = client->GetBalance(account.Address(), "NEO");
        std::cout << "Current balance: " << balance << " NEO" << std::endl;
        
        if (balance < 10) {
            std::cout << "Insufficient balance!" << std::endl;
            return 1;
        }
        
        // Build transaction
        std::cout << "Building transaction..." << std::endl;
        auto tx = neo::sdk::TransactionBuilder()
            .From(account)
            .To(recipient)
            .Amount(10, "NEO")
            .WithGas(1000000, 500000)  // System fee, Network fee
            .Sign(account.GetPrivateKey())
            .Build();
        
        // Send transaction
        std::cout << "Sending transaction..." << std::endl;
        auto txHash = client->SendRawTransaction(tx);
        
        std::cout << "Transaction sent successfully!" << std::endl;
        std::cout << "TX Hash: " << txHash.ToString() << std::endl;
        std::cout << "View on explorer: https://neoscan.io/transaction/" 
                  << txHash.ToString() << std::endl;
        
        // Wait for confirmation
        std::cout << "Waiting for confirmation..." << std::endl;
        if (client->WaitForTransaction(txHash, 30000)) {
            std::cout << "Transaction confirmed!" << std::endl;
            
            // Check new balance
            auto newBalance = client->GetBalance(account.Address(), "NEO");
            std::cout << "New balance: " << newBalance << " NEO" << std::endl;
        }
        
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
    
    return 0;
}

Checking Account Balances

Query balances for NEO, GAS, and NEP-17 tokens.

#include <neo/sdk/sdk.h>
#include <iostream>
#include <iomanip>

void CheckBalance(neo::sdk::RpcClient& client, const std::string& address) {
    std::cout << "\nBalance for: " << address << std::endl;
    std::cout << std::string(50, '-') << std::endl;
    
    // Check NEO balance
    auto neoBalance = client.GetBalance(address, "NEO");
    std::cout << "NEO:  " << std::setw(15) << neoBalance << " NEO" << std::endl;
    
    // Check GAS balance
    auto gasBalance = client.GetBalance(address, "GAS");
    std::cout << "GAS:  " << std::setw(15) << std::fixed 
              << std::setprecision(8) << gasBalance / 100000000.0 << " GAS" << std::endl;
    
    // Check NEP-17 tokens
    auto tokens = client.GetNEP17Balances(address);
    
    if (!tokens.empty()) {
        std::cout << "\nNEP-17 Tokens:" << std::endl;
        
        for (const auto& token : tokens) {
            neo::sdk::NEP17Token nep17(token.AssetHash, client);
            
            auto symbol = nep17.Symbol();
            auto decimals = nep17.Decimals();
            auto balance = token.Amount;
            
            double displayBalance = balance / std::pow(10, decimals);
            
            std::cout << symbol << ": " << std::setw(15) 
                     << std::fixed << std::setprecision(decimals) 
                     << displayBalance << " " << symbol << std::endl;
        }
    }
}

int main() {
    try {
        neo::sdk::RpcClient client("http://seed1.neo.org:20332");
        
        // List of addresses to check
        std::vector<std::string> addresses = {
            "NXjtqYERuvSWGawjVux8UerNejvwdYg7eE",
            "NTrezR3C4X8aMLVg7vozt5wguyNfFhwuFx",
            "NPTmAHDxo6Pkyic8Nvu3kwyXoYJCvcCB6Y"
        };
        
        for (const auto& address : addresses) {
            CheckBalance(client, address);
        }
        
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
    
    return 0;
}

Smart Contract Deployment and Invocation

Deploy a smart contract and interact with it.

#include <neo/sdk/sdk.h>
#include <iostream>
#include <fstream>

int main() {
    try {
        auto client = std::make_shared<neo::sdk::RpcClient>("http://localhost:20332");
        auto wallet = neo::sdk::Wallet::Load("wallet.json", "password");
        auto account = wallet.GetDefaultAccount();
        
        // Deploy contract
        std::cout << "Deploying smart contract..." << std::endl;
        
        neo::sdk::ContractDeployer deployer;
        deployer.LoadNEF("mycontract.nef");
        deployer.LoadManifest("mycontract.manifest.json");
        
        auto deployment = deployer
            .WithSigner(account)
            .WithGas(10000000)  // 0.1 GAS
            .Deploy(client);
        
        std::cout << "Contract deployed!" << std::endl;
        std::cout << "Contract Hash: " << deployment.ContractHash.ToString() << std::endl;
        std::cout << "Transaction: " << deployment.TransactionHash.ToString() << std::endl;
        
        // Wait for deployment
        client->WaitForTransaction(deployment.TransactionHash, 60000);
        
        // Create contract interface
        neo::sdk::Contract contract(deployment.ContractHash, client);
        
        // Call read-only method
        std::cout << "\nCalling getName()..." << std::endl;
        auto name = contract.Call("getName");
        std::cout << "Contract Name: " << name.GetString() << std::endl;
        
        // Call state-changing method
        std::cout << "\nCalling setValue(42)..." << std::endl;
        auto result = contract
            .Invoke("setValue")
            .WithArgs(42)
            .WithSigner(account)
            .WithGas(1000000)
            .Send();
        
        std::cout << "Transaction: " << result.TransactionHash.ToString() << std::endl;
        
        // Subscribe to events
        contract.OnEvent("ValueChanged", [](const neo::sdk::ContractEvent& event) {
            auto oldValue = event.GetArg<int>(0);
            auto newValue = event.GetArg<int>(1);
            auto sender = event.GetArg<std::string>(2);
            
            std::cout << "Event: Value changed from " << oldValue 
                     << " to " << newValue 
                     << " by " << sender << std::endl;
        });
        
        // Start event listener
        contract.StartEventListener();
        
        std::cout << "\nListening for events (press Ctrl+C to stop)..." << std::endl;
        
        // Keep listening
        while (true) {
            std::this_thread::sleep_for(std::chrono::seconds(1));
        }
        
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
    
    return 0;
}

Complete Token Transfer Application

A full-featured CLI application for managing and transferring tokens.

// Download the complete example
// https://github.com/r3e-network/neo_cpp/tree/main/examples/token-transfer

#include <neo/sdk/sdk.h>
#include <iostream>
#include <string>
#include <map>
#include <functional>

class TokenTransferCLI {
private:
    std::shared_ptr<neo::sdk::RpcClient> client;
    std::unique_ptr<neo::sdk::Wallet> wallet;
    std::map<std::string, std::function<void()>> commands;
    
public:
    TokenTransferCLI() {
        InitializeCommands();
    }
    
    void InitializeCommands() {
        commands["help"] = [this]() { ShowHelp(); };
        commands["connect"] = [this]() { Connect(); };
        commands["wallet"] = [this]() { ManageWallet(); };
        commands["balance"] = [this]() { ShowBalance(); };
        commands["transfer"] = [this]() { Transfer(); };
        commands["history"] = [this]() { ShowHistory(); };
        commands["quit"] = [this]() { std::exit(0); };
    }
    
    void Run() {
        std::cout << "Neo Token Transfer CLI v1.0" << std::endl;
        std::cout << "Type 'help' for commands" << std::endl << std::endl;
        
        std::string command;
        while (true) {
            std::cout << "neo> ";
            std::cin >> command;
            
            if (commands.find(command) != commands.end()) {
                try {
                    commands[command]();
                } catch (const std::exception& e) {
                    std::cerr << "Error: " << e.what() << std::endl;
                }
            } else {
                std::cout << "Unknown command. Type 'help' for available commands." << std::endl;
            }
        }
    }
    
    void ShowHelp() {
        std::cout << "\nAvailable commands:" << std::endl;
        std::cout << "  help     - Show this help message" << std::endl;
        std::cout << "  connect  - Connect to Neo network" << std::endl;
        std::cout << "  wallet   - Manage wallet" << std::endl;
        std::cout << "  balance  - Show account balances" << std::endl;
        std::cout << "  transfer - Transfer tokens" << std::endl;
        std::cout << "  history  - Show transaction history" << std::endl;
        std::cout << "  quit     - Exit application" << std::endl << std::endl;
    }
    
    void Connect() {
        std::string url;
        std::cout << "Enter RPC URL (default: http://seed1.neo.org:20332): ";
        std::cin.ignore();
        std::getline(std::cin, url);
        
        if (url.empty()) {
            url = "http://seed1.neo.org:20332";
        }
        
        client = std::make_shared<neo::sdk::RpcClient>(url);
        
        auto height = client->GetBlockCount();
        std::cout << "Connected! Current height: " << height << std::endl;
    }
    
    // ... Additional methods implementation ...
};

int main() {
    TokenTransferCLI app;
    app.Run();
    return 0;
}

Building the Examples

Prerequisites

  • Neo C++ SDK installed
  • C++20 compatible compiler
  • CMake 3.20+

Build All Examples

# Clone the repository
git clone https://github.com/r3e-network/neo_cpp.git
cd neo-cpp/examples

# Build with CMake
cmake -B build
cmake --build build

# Run an example
./build/hello_neo

CMakeLists.txt

cmake_minimum_required(VERSION 3.20)
project(NeoExamples)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Neo SDK
find_package(NeoSDK REQUIRED)

# Add examples
add_executable(hello_neo hello_neo.cpp)
target_link_libraries(hello_neo Neo::SDK)

add_executable(wallet_creation wallet_creation.cpp)
target_link_libraries(wallet_creation Neo::SDK)
