Quick Start

Get up and running in minutes

# Pull the Docker image
docker pull neocpp/node:latest

# Run a testnet node
docker run -d \
  --name neo-node \
  -p 10333:10333 \
  -p 10334:10334 \
  -v neo-data:/data \
  neocpp/node:latest --testnet

# Check node status
docker exec neo-node neo-cli state
# Clone the repository
git clone https://github.com/r3e-network/neo_cpp.git
cd neo-cpp

# Build with CMake
cmake -B build -DCMAKE_BUILD_TYPE=Release
cmake --build build --parallel

# Run the node
./build/apps/neo_node --config testnet.json
# Download the latest release
wget https://github.com/r3e-network/neo_cpp/releases/latest/download/neo-cpp-linux.tar.gz

# Extract the archive
tar -xzf neo-cpp-linux.tar.gz
cd neo-cpp

# Run the node
./neo_node --testnet

Code Examples

Learn by example with real-world use cases

Create a Wallet

#include <neo/sdk/wallet.h>

// Create a new wallet
neo::Wallet wallet;
wallet.Create("MyWallet");

// Generate new account
auto account = wallet.CreateAccount();
std::cout << "Address: " << account.Address() << std::endl;

// Save wallet
wallet.Save("wallet.json", "password");

Send Transaction

#include <neo/sdk/transaction.h>

// Build transaction
neo::TransactionBuilder tx;
tx.From(account)
  .To("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE")
  .Amount(100, neo::GAS)
  .Sign(privateKey);

// Send to network
auto hash = client.SendTransaction(tx);
std::cout << "TX: " << hash << std::endl;

Deploy Smart Contract

#include <neo/sdk/contract.h>

// Load contract
neo::Contract contract;
contract.LoadNEF("contract.nef");
contract.LoadManifest("contract.manifest.json");

// Deploy to blockchain
auto deployment = contract.Deploy()
  .WithSigner(account)
  .WithGas(1000)
  .Send(client);

std::cout << "Contract: " << deployment.Hash() << std::endl;

Query Blockchain

#include <neo/sdk/rpc.h>

// Connect to RPC
neo::RpcClient client("http://localhost:10334");

// Get blockchain info
auto height = client.GetBlockCount();
auto block = client.GetBlock(height - 1);

std::cout << "Height: " << height << std::endl;
std::cout << "TX Count: " << block.Transactions.size() << std::endl;
