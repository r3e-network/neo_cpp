Neo C++ SDK Guide

Complete guide to building blockchain applications with the Neo C++ SDK

Introduction

The Neo C++ SDK provides a comprehensive toolkit for building blockchain applications. It offers high-level abstractions for wallet management, transaction building, smart contract interaction, and blockchain querying.

SDK Features:
  • NEP-6 compliant wallet management
  • Transaction building and signing
  • Smart contract deployment and invocation
  • NEP-17 token standard support
  • Full RPC client implementation
  • Event subscription and handling
  • Cryptographic utilities

Installation

Using CMake

# CMakeLists.txt
cmake_minimum_required(VERSION 3.20)
project(MyDApp)

# Find Neo SDK
find_package(NeoSDK REQUIRED)

# Create executable
add_executable(mydapp main.cpp)

# Link Neo SDK
target_link_libraries(mydapp PRIVATE Neo::SDK)

Manual Installation

# Build and install SDK
cd neo-cpp
cmake -B build -DBUILD_SDK=ON
cmake --build build
sudo cmake --install build

# SDK will be installed to:
# Headers: /usr/local/include/neo/sdk/
# Library: /usr/local/lib/libneo-sdk.a

Quick Start

Complete Example: Send NEO

#include 
#include 

int main() {
    try {
        // 1. Connect to Neo network
        auto client = std::make_shared("http://seed1.neo.org:20332");
        
        // 2. Create or load wallet
        neo::sdk::Wallet wallet("MyWallet");
        auto account = wallet.CreateAccount("Main");
        
        std::cout << "Created account: " << account.Address() << std::endl;
        
        // 3. Check balance
        auto balance = client->GetBalance(account.Address(), "NEO");
        std::cout << "Balance: " << balance << " NEO" << std::endl;
        
        // 4. Build transaction
        neo::sdk::TransactionBuilder builder;
        auto tx = builder
            .From(account)
            .To("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE")
            .Amount(10, "NEO")
            .WithGas(1000000, 500000)
            .Sign(account.GetPrivateKey())
            .Build();
        
        // 5. Send transaction
        auto txHash = client->SendRawTransaction(tx);
        std::cout << "Transaction sent: " << txHash.ToString() << std::endl;
        
        // 6. Wait for confirmation
        client->WaitForTransaction(txHash, 30000); // 30 seconds timeout
        std::cout << "Transaction confirmed!" << std::endl;
        
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
    
    return 0;
}

Wallet Management

Creating a Wallet

// Create new wallet
neo::sdk::Wallet wallet("MyWallet");

// Create accounts
auto account1 = wallet.CreateAccount("Primary Account");

wallet.SetDefaultAccount(account1);

// Save encrypted wallet
wallet.Save("wallet.json", "strongPassword123");

Loading a Wallet

// Load existing wallet

Building Transactions

Simple Transfer

// Transfer NEO
auto tx = neo::sdk::TransactionBuilder()
    .From(account)
    .To("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE")
    .Amount(100, "NEO")
    .Sign(account.GetPrivateKey())
    .Build();

Multiple Transfers

// Send to multiple recipients
auto builder = neo::sdk::TransactionBuilder()
    .From(account);

// Add multiple outputs
builder.Transfer("Address1", 10, "NEO")
       .Transfer("Address2", 20, "GAS")
       .Transfer("Address3", 30, "NEO");

// Build and sign
auto tx = builder.Sign(account.GetPrivateKey()).Build();

Custom Scripts

// Build custom script
neo::sdk::ScriptBuilder sb;
sb.EmitPush(100)
  .EmitPush(account.GetScriptHash())
  .EmitPush("transfer")
  .EmitPush(contractHash)
  .EmitSysCall(neo::sdk::InteropService::System_Contract_Call);

// Create transaction with script
auto tx = neo::sdk::TransactionBuilder()
    .WithScript(sb.ToArray())
    .WithSigner(account.GetScriptHash(), WitnessScope::CalledByEntry)
    .WithGas(1000000, 500000)
    .Sign(account.GetPrivateKey())
    .Build();

Transaction Attributes

// Add attributes
auto tx = neo::sdk::TransactionBuilder()
    .From(account)
    .To(recipient)
    .Amount(100, "NEO")
    .WithAttribute(TransactionAttributeType::HighPriority)
    .WithValidUntilBlock(currentHeight + 100)
    .WithSystemFee(1000000)
    .WithNetworkFee(500000)
    .Sign(account.GetPrivateKey())
    .Build();

Smart Contract Interaction

Deploying a Contract

// Load contract files
neo::sdk::ContractDeployer deployer;
deployer.LoadNEF("mycontract.nef");
deployer.LoadManifest("mycontract.manifest.json");

// Deploy contract
auto deployment = deployer
    .WithSigner(account)
    .WithGas(10000000)
    .Deploy(client);

std::cout << "Contract deployed: " << deployment.ContractHash << std::endl;
std::cout << "Transaction: " << deployment.TransactionHash << std::endl;

Invoking Contract Methods

// Create contract interface
neo::sdk::Contract contract(contractHash, client);

// Call read-only method
auto result = contract.Call("balanceOf", account.GetScriptHash());
std::cout << "Balance: " << result.GetInteger() << std::endl;

// Invoke state-changing method
auto invocation = contract
    .Invoke("transfer")
    .WithArgs(fromAddress, toAddress, amount, null)
    .WithSigner(account)
    .WithGas(1000000)
    .Send();

std::cout << "TX: " << invocation.TransactionHash << std::endl;

Contract Events

// Subscribe to contract events
contract.OnEvent("Transfer", [](const neo::sdk::ContractEvent& event) {
    auto from = event.GetArg(0);
    auto to = event.GetArg(1);
    auto amount = event.GetArg(2);
    
    std::cout << "Transfer: " << from << " -> " << to 
              << " Amount: " << amount << std::endl;
});

// Start listening
contract.StartEventListener();

Custom Token Implementation

class MyToken : public neo::sdk::NEP17Token {
public:
    MyToken(const UInt160& scriptHash, RpcClient& client)
        : NEP17Token(scriptHash, client) {}
    
    // Add custom methods
    TransactionBuilder Mint(const Account& owner, const BigInteger& amount) {
        return Invoke("mint")
            .WithArgs(owner.GetScriptHash(), amount)
            .WithSigner(owner);
    }
    
    TransactionBuilder Burn(const Account& owner, const BigInteger& amount) {
        return Invoke("burn")
            .WithArgs(owner.GetScriptHash(), amount)
            .WithSigner(owner);
    }
};

Event Handling

Block Events

// Subscribe to new blocks
client.OnNewBlock([](const neo::sdk::Block& block) {
    std::cout << "New Block: " << block.Index << std::endl;
    std::cout << "Hash: " << block.Hash << std::endl;
    std::cout << "Transactions: " << block.Transactions.size() << std::endl;
});

// Start listening
client.StartBlockListener();

Transaction Events

// Monitor specific address
client.OnTransaction(address, [](const neo::sdk::Transaction& tx) {
    std::cout << "Transaction for address detected!" << std::endl;
    std::cout << "TX Hash: " << tx.Hash << std::endl;
    
    // Process transaction
    for (const auto& transfer : tx.GetTransfers()) {
        if (transfer.To == address) {
            std::cout << "Received: " << transfer.Amount 
                     << " " << transfer.Asset << std::endl;
        }
    }
});

WebSocket Subscriptions

// WebSocket client for real-time updates
neo::sdk::WebSocketClient ws("ws://localhost:10334");

// Subscribe to events
ws.Subscribe("block_added");
ws.Subscribe("transaction_added");
ws.Subscribe("notification_added");

// Handle messages
ws.OnMessage([](const json& msg) {
    auto event = msg["event"].get();
    
    if (event == "block_added") {
        auto blockHeight = msg["data"]["height"].get();
        std::cout << "New block: " << blockHeight << std::endl;
    }
});

ws.Connect();

Security Best Practices

Key Management

// Secure key generation
auto privateKey = neo::sdk::PrivateKey::Generate();

// Encrypt private key
auto encrypted = privateKey.Encrypt("password", 
    neo::sdk::ScryptParameters::Default);

// Store securely
SecureStorage::Store("key_id", encrypted);

// Decrypt when needed
auto decrypted = neo::sdk::PrivateKey::Decrypt(encrypted, "password");

Transaction Validation

// Always validate before sending
bool ValidateTransaction(const Transaction& tx) {
    // Check signatures
    if (!tx.VerifyWitnesses()) {
        std::cerr << "Invalid signatures" << std::endl;
        return false;
    }
    
    // Check fees
    if (tx.SystemFee + tx.NetworkFee > MAX_ACCEPTABLE_FEE) {
        std::cerr << "Fees too high" << std::endl;
        return false;
    }
    
    // Validate addresses
    for (const auto& signer : tx.Signers) {
        if (!IsValidAddress(signer.Account)) {
            std::cerr << "Invalid signer address" << std::endl;
            return false;
        }
    }
    
    return true;
}
Security Warning:
  • Never store private keys in plain text
  • Always validate user inputs
  • Use secure random number generation
  • Implement rate limiting for RPC calls
  • Audit smart contracts before deployment

Integration Testing

// Test with local private net
class IntegrationTest : public ::testing::Test {
protected:
    void SetUp() override {
        // Start local blockchain
        blockchain = std::make_unique();
        blockchain->Start();
        
        // Connect client
        client = std::make_unique("http://localhost:30332");
    }
    
    void TearDown() override {
        blockchain->Stop();
    }
    
    std::unique_ptr blockchain;
    std::unique_ptr client;
};

TEST_F(IntegrationTest, TransferNEO) {
    // Test actual transfer on private net
    auto wallet = neo::sdk::Wallet::Load("test-wallet.json", "password");
    auto account = wallet.GetDefaultAccount();
    
    auto initialBalance = client->GetBalance(account.Address(), "NEO");
    
    // Send transaction
    auto tx = neo::sdk::TransactionBuilder()
        .From(account)
        .To("TestAddress")
        .Amount(10, "NEO")
        .Sign(account.GetPrivateKey())
        .Build();
    
    client->SendRawTransaction(tx);
    blockchain->ProduceBlock();
    
    auto finalBalance = client->GetBalance(account.Address(), "NEO");
    EXPECT_EQ(finalBalance, initialBalance - 10);
}

Example: Token Transfer dApp

// Complete token transfer application
#include 
#include 
#include 

class TokenTransferApp {
private:
    std::shared_ptr client;
    std::unique_ptr wallet;
    
public:
    TokenTransferApp(const std::string& rpcUrl) {
        client = std::make_shared(rpcUrl);
    }
    
    void LoadWallet(const std::string& path, const std::string& password) {
        wallet = std::make_unique(
            neo::sdk::Wallet::Load(path, password)
        );
        std::cout << "Wallet loaded with " << wallet->GetAccounts().size() 
                  << " accounts" << std::endl;
    }
    
    void ShowBalances() {
        for (const auto& account : wallet->GetAccounts()) {
            auto neoBalance = client->GetBalance(account.Address(), "NEO");
            auto gasBalance = client->GetBalance(account.Address(), "GAS");
            
            std::cout << "\nAccount: " << account.Address() << std::endl;
            std::cout << "  NEO: " << neoBalance << std::endl;
            std::cout << "  GAS: " << gasBalance << std::endl;
        }
    }
    
    void Transfer(const std::string& to, int64_t amount, const std::string& asset) {
        auto account = wallet->GetDefaultAccount();
        
        std::cout << "\nTransferring " << amount << " " << asset 
                  << " to " << to << std::endl;
        
        try {
            auto tx = neo::sdk::TransactionBuilder()
                .From(account)
                .To(to)
                .Amount(amount, asset)
                .WithGas(1000000, 500000)
                .Sign(account.GetPrivateKey())
                .Build();
            
            auto txHash = client->SendRawTransaction(tx);
            std::cout << "Transaction sent: " << txHash.ToString() << std::endl;
            
            // Wait for confirmation
            std::cout << "Waiting for confirmation..." << std::endl;
            client->WaitForTransaction(txHash, 60000);
            std::cout << "Transaction confirmed!" << std::endl;
            
        } catch (const std::exception& e) {
            std::cerr << "Transfer failed: " << e.what() << std::endl;
        }
    }
    
    void Run() {
        std::string command;
        
        while (true) {
            std::cout << "\n> ";
            std::cin >> command;
            
            if (command == "balance") {
                ShowBalances();
            } else if (command == "transfer") {
                std::string to, asset;
                int64_t amount;
                
                std::cout << "To address: ";
                std::cin >> to;
                std::cout << "Amount: ";
                std::cin >> amount;
                std::cout << "Asset (NEO/GAS): ";
                std::cin >> asset;
                
                Transfer(to, amount, asset);
            } else if (command == "quit") {
                break;
            } else {
                std::cout << "Commands: balance, transfer, quit" << std::endl;
            }
        }
    }
};

int main() {
    TokenTransferApp app("http://seed1.neo.org:20332");
    app.LoadWallet("wallet.json", "password");
    app.Run();
    return 0;
}
