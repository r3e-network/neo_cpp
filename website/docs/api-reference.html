Complete API documentation for Neo C++ SDK and core components

neo::sdk::Wallet

Manages NEP-6 compliant wallets with account management and transaction signing.

Include

#include <neo/sdk/wallet/wallet.h>

Constructor

Wallet();
Wallet(const std::string& name);
Wallet(const std::filesystem::path& path, const std::string& password);

Methods

Create()

void Create(const std::string& name);

Creates a new wallet with the specified name.

CreateAccount()

Account CreateAccount();
Account CreateAccount(const std::string& label);

Creates a new account in the wallet with an optional label.

ImportAccount()

Account ImportAccount(const std::string& wif);
Account ImportAccount(const PrivateKey& privateKey);

Imports an account from WIF or private key.

Save()

void Save(const std::filesystem::path& path, const std::string& password);

Saves the wallet to a file encrypted with the password.

Load()

static Wallet Load(const std::filesystem::path& path, const std::string& password);

Loads a wallet from an encrypted file.

GetAccounts()

std::vector<Account> GetAccounts() const;

Returns all accounts in the wallet.

GetDefaultAccount()

Account GetDefaultAccount() const;

Returns the default account of the wallet.

Example

// Create a new wallet
neo::sdk::Wallet wallet("MyWallet");

// Create a new account
auto account = wallet.CreateAccount("Main Account");
std::cout << "Address: " << account.Address() << std::endl;

// Import from WIF
auto imported = wallet.ImportAccount("L1QqQJnpBwbsPGAuutuzPTac8piqvbR1HRjrY5qHup48TBCBFe4g");

// Save encrypted wallet
wallet.Save("wallet.json", "strongPassword123");

// Load existing wallet
auto loaded = Wallet::Load("wallet.json", "strongPassword123");

neo::sdk::TransactionBuilder

Builds and signs blockchain transactions.

Include

#include <neo/sdk/tx/transaction_builder.h>

Methods

From()

TransactionBuilder& From(const Account& account);

Sets the sender account for the transaction.

To()

TransactionBuilder& To(const std::string& address);

Sets the recipient address.

Amount()

TransactionBuilder& Amount(int64_t amount, const std::string& asset = "NEO");

Sets the transfer amount and asset type.

WithScript()

TransactionBuilder& WithScript(const std::vector<uint8_t>& script);

Adds a custom script to the transaction.

WithGas()

TransactionBuilder& WithGas(int64_t systemFee, int64_t networkFee);

Sets the gas fees for the transaction.

Sign()

TransactionBuilder& Sign(const PrivateKey& key);

Signs the transaction with the provided private key.

Build()

Transaction Build();

Builds and returns the complete transaction.

Example

// Build a NEO transfer transaction
auto tx = neo::sdk::TransactionBuilder()
    .From(account)
    .To("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE")
    .Amount(100, "NEO")
    .WithGas(1000000, 500000)
    .Sign(account.GetPrivateKey())
    .Build();

// Send via RPC
rpcClient.SendRawTransaction(tx);

Example

// Connect to testnet
neo::sdk::RpcClient client("http://seed1.neo.org:20332");

// Get blockchain info
auto height = client.GetBlockCount();
std::cout << "Current height: " << height << std::endl;

// Get latest block
auto block = client.GetBlock(height - 1);
std::cout << "Block hash: " << block.Hash.ToString() << std::endl;
std::cout << "Transactions: " << block.Transactions.size() << std::endl;

// Check balance
auto balance = client.GetBalance("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE", "NEO");
std::cout << "Balance: " << balance << " NEO" << std::endl;

neo::sdk::ContractInvoker

Invokes and deploys smart contracts.

Include

#include <neo/sdk/contract/contract_invoker.h>

Methods

LoadNEF()

void LoadNEF(const std::filesystem::path& path);

Loads a NEF (Neo Executable Format) file.

LoadManifest()

void LoadManifest(const std::filesystem::path& path);

Loads a contract manifest file.

Deploy()

DeploymentBuilder Deploy();

Creates a deployment builder for the contract.

Invoke()

InvocationBuilder Invoke(const std::string& method);

Creates an invocation builder for a contract method.

Example

// Deploy a smart contract
neo::sdk::ContractInvoker contract;
contract.LoadNEF("mycontract.nef");
contract.LoadManifest("mycontract.manifest.json");

auto deployment = contract.Deploy()
    .WithSigner(account)
    .WithGas(10000000)
    .Send(rpcClient);

std::cout << "Contract deployed: " << deployment.ContractHash << std::endl;

// Invoke contract method
auto result = contract.Invoke("transfer")
    .WithArgs(fromAddress, toAddress, amount)
    .WithSigner(account)
    .Send(rpcClient);

std::cout << "TX: " << result.TransactionHash << std::endl;

neo::sdk::NEP17Token

Interface for NEP-17 standard tokens (fungible tokens).

Include

#include <neo/sdk/contract/nep17_token.h>

Constructor

NEP17Token(const UInt160& scriptHash, RpcClient& client);

Methods

Symbol()

std::string Symbol();

Returns the token symbol.

Decimals()

uint8_t Decimals();

Returns the number of decimals.

TotalSupply()

BigInteger TotalSupply();

Returns the total token supply.

BalanceOf()

BigInteger BalanceOf(const std::string& address);

Returns the token balance of an address.

Transfer()

TransactionBuilder Transfer(const Account& from, const std::string& to, const BigInteger& amount);

Creates a transfer transaction.

Example

// Connect to GAS token
neo::sdk::RpcClient client("http://localhost:10334");
neo::sdk::NEP17Token gas(GAS_SCRIPT_HASH, client);

// Get token info
std::cout << "Symbol: " << gas.Symbol() << std::endl;
std::cout << "Decimals: " << (int)gas.Decimals() << std::endl;
std::cout << "Total Supply: " << gas.TotalSupply() << std::endl;

// Check balance
auto balance = gas.BalanceOf("NXjtqYERuvSWGawjVux8UerNejvwdYg7eE");
std::cout << "Balance: " << balance << std::endl;

// Transfer tokens
auto tx = gas.Transfer(account, "NTrezR3C4X8aMLVg7vozt5wguyNfFhwuFx", 1000000000)
    .Sign(account.GetPrivateKey())
    .Build();

client.SendRawTransaction(tx);

#include <neo/cryptography/crypto.h>

Double SHA256 hash.

UInt160 Hash160(const std::vector<uint8_t>& data);

SHA256 followed by RIPEMD160.

ECDSA

Sign

std::vector<uint8_t> Sign(const std::vector<uint8_t>& message, 
                          const PrivateKey& key);

Verify

bool Verify(const std::vector<uint8_t>& message, 
            const std::vector<uint8_t>& signature, 
            const ECPoint& publicKey);

Encoding

Base58

std::string Base58Encode(const std::vector<uint8_t>& data);
std::vector<uint8_t> Base58Decode(const std::string& encoded);

Base64

std::string Base64Encode(const std::vector<uint8_t>& data);
std::vector<uint8_t> Base64Decode(const std::string& encoded);

Example

// Hash operations
auto hash = neo::crypto::SHA256("Hello Neo");
auto hash160 = neo::crypto::Hash160(publicKey.ToArray());

// ECDSA signing
neo::crypto::PrivateKey privateKey;
auto signature = neo::crypto::Sign(message, privateKey);

// Verification
auto publicKey = privateKey.GetPublicKey();
bool valid = neo::crypto::Verify(message, signature, publicKey);

// Encoding
auto encoded = neo::crypto::Base58Encode(data);
auto decoded = neo::crypto::Base58Decode(encoded);
